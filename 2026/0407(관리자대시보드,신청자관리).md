- 진행 중인 방송 리스트
    - tb_broadcast_setting [일정방송] - 제외
    - tb_broadcast_msg_send [음성방송] - `result`
    - tb_user_msg_send [메시지방송] - `result`
    - tb_user_msg_lms_send [장문메시지방송] - `result`
```sql
SELECT 
    msg_category,
    dispatch_group_no,
    MIN(transmission_requested_at) AS first_request_at,
    MAX(transmission_completed_at) AS transmission_completed_at,
    COUNT(*) AS total_cnt,
    SUM(CASE WHEN detail_result = '0' THEN 1 ELSE 0 END) AS success_cnt,
    SUM(CASE WHEN detail_result != '0' AND detail_result IS NOT NULL THEN 1 ELSE 0 END) AS fail_cnt,
    SUM(CASE WHEN detail_result IS NULL THEN 1 ELSE 0 END) AS processing_cnt,
    ROUND(SUM(CASE WHEN detail_result = '0' THEN 1 ELSE 0 END) * 100.0 / COUNT(*), 2) AS delivery_rate,
    CASE 
        WHEN COUNT(*) = SUM(CASE WHEN detail_result IS NOT NULL THEN 1 ELSE 0 END) THEN 'COMPLETED'
        ELSE 'PROCESSING'
    END AS send_status
FROM (
    SELECT 'USER_MSG' AS msg_category, dispatch_group_no, detail_result, transmission_requested_at, transmission_completed_at
    FROM tb_user_msg_send 
    WHERE transmission_requested_at >= CURDATE()
    UNION ALL
    SELECT 'BROADCAST' AS msg_category, dispatch_group_no, detail_result, transmission_requested_at, transmission_completed_at
    FROM tb_broadcast_msg_send 
    WHERE transmission_requested_at >= CURDATE()
    UNION ALL
    SELECT 'LMS_MSG' AS msg_category, dispatch_group_no, detail_result, transmission_requested_at, transmission_completed_at
    FROM tb_user_msg_lms_send 
    WHERE transmission_requested_at >= CURDATE()
) AS combined_data
GROUP BY msg_category, dispatch_group_no
ORDER BY first_request_at DESC;
```
